Federated exchanges allow for messages published to an exchange on an upstream node to transparently be published to an exchange of the same name on the downstream node.




rabbitmq-plugins enable rabbitmq_federation
rabbitmq-plugins enable rabbitmq_federation_management


# up node, the exchange name should be identical with the down node
rabbitmqadmin declare exchange name=demo-exchange type=direct
rabbitmqadmin declare queue name=qq1 arguments='{"x-queue-type": "quorum"}'
rabbitmqadmin declare binding source=demo-exchange destination=qq1 routing_key=qq1

# down node
rabbitmqadmin declare exchange name=demo-exchange type=direct
rabbitmqadmin declare queue name=qq1 arguments='{"x-queue-type": "quorum"}'
rabbitmqadmin declare binding source=demo-exchange destination=qq1 routing_key=qq1


# configure the federation in the down node
rabbitmqctl set_parameter federation-upstream fed-demo '{"uri":"amqp://admin:password@X.X.X.X:5672"}'
rabbitmqctl set_policy policy-demo "^demo-exchange" '{"federation-upstream":"fed-demo"}' --apply-to=exchanges



# publish the message in the up node
for i in $(seq 100); do rabbitmqadmin publish exchange=demo-exchange routing_key="qq1" payload="hello, world  $i"; done


the message will be sync from up to down





# consume the message
rabbitmqadmin get queue=qq1 ackmode=ack_requeue_true count=1000
