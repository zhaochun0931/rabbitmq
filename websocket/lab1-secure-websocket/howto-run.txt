# Create a user named 'mqtt_test' with the password 'secret'
rabbitmqctl add_user mqtt_test secret

# Make the user an administrator (optional, but easy for testing)
rabbitmqctl set_user_tags mqtt_test administrator

# Grant the user full read/write permissions on the default virtual host ("/")
rabbitmqctl set_permissions -p / mqtt_test ".*" ".*" ".*"
