

rabbitmq-plugins enable rabbitmq_federation
rabbitmq-plugins enable rabbitmq_federation_management


# up
rabbitmqadmin declare exchange name=up-exchange type=direct
rabbitmqadmin declare queue name=qq-up arguments='{"x-queue-type": "quorum"}'
rabbitmqadmin declare binding source=fed-up-demo destination=qq-up routing_key=qq-up

# down
rabbitmqadmin declare exchange name=down-exchange type=direct
rabbitmqadmin declare queue name=qq-down arguments='{"x-queue-type": "quorum"}'
rabbitmqadmin declare binding source=fed-up-demo destination=qq-down routing_key=qq-down



rabbitmqctl set_parameter federation-upstream fed-demo '{"uri":"amqp://admin:password@localhost:5672"}'
rabbitmqctl set_policy policy-demo "^down" '{"federation-upstream":"fed-demo"}' --apply-to exchange
